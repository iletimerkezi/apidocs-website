function xmlpost(url, postdata: string): String;
var
IdHTTP1: TIdHTTP;
str1:TStringList;
begin
  try
    try
      IdHTTP1:=TIdHTTP.Create(nil);
      str1:=TStringList.Create;
      str1.Text:=postdata;
      result:=IdHTTP1.Post(url, str1);
    except
    end;
  finally
    FreeAndNil(str1);
    FreeAndNil(IdHTTP1);
  end;
end;
